#include <bits/stdc++.h>
using namespace std;
int main()
{
    unordered_map<int, int> mp;
    mp.insert(pair<int, int>(2, 20));
    mp.insert(pair<int, int>(3, 10));
    mp[4] = 80;
    cout << mp.size() << endl; // 3

    // mp.clear(); - to make the map empty

    if (mp.empty())    // Checking if Map is empty
        cout << "Yes" << endl;
    else
        cout << "No" << endl;    // No

    for (auto it = mp.begin(); it != mp.end(); it++)    // Iterating over Map
        cout << it->first << " " << it->second << endl;

        // Result: 
        //         4 80 
        //         3 10 
        //         2 20

    return 0;
}
