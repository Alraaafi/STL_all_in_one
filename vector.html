#include <bits/stdc++.h>
using namespace std;
int main()
{
    vector<int> v; // vector declaration
    v.push_back(2);
    v.push_back(4);
    v.push_back(6);
    cout << v.size() << endl;   // 3

    for (int i = 0; i < v.size(); i++)
        cout << v[i] << " ";    // 2 4 6
    cout << endl;

    v.pop_back();   // Pops/Removes Item from back

    // iterator
    for (auto it = v.begin(); it != v.end(); it++)
        cout << *it << " ";     // 2 4
    cout << endl;
}
